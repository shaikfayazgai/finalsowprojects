```typescript
'use client'
import { useState } from 'react'
import { useRouter } from 'next/navigation'
import { Button, TextInput } from '@glimmora/ui'

export function VerifyEmailStep() {
  const router = useRouter()
  const [code, setCode] = useState('')
  const [isVerifying, setIsVerifying] = useState(false)
  const [sent, setSent] = useState(false)

  async function handleSendCode() {
    await fetch('/api/auth/verify-email', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ action: 'send' }) })
    setSent(true)
  }

  async function handleVerify() {
    setIsVerifying(true)
    try {
      await fetch('/api/auth/verify-email', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ action: 'verify', code }) })
      router.push('/onboarding/skills')
    } finally {
      setIsVerifying(false)
    }
  }

  return (
    <div className="flex flex-col gap-5">
      <h2 className="font-display text-xl font-semibold text-text-heading">Verify Your University Email</h2>
      <p className="text-sm font-body text-text-caption">We sent a verification code to your university email address. Enter it below to continue.</p>

      {!sent ? (
        <Button variant="primary" className="w-full" onClick={handleSendCode}>
          Send Verification Code
        </Button>
      ) : (
        <>
          <TextInput
            label="Verification Code"
            placeholder="Enter 6-digit code"
            value={code}
            onChange={(e) => setCode(e.target.value)}
            className="text-center text-lg tracking-widest"
          />
          <Button variant="primary" className="w-full" onClick={handleVerify} disabled={isVerifying || code.length < 6}>
            {isVerifying ? 'Verifying...' : 'Verify Email'}
          </Button>
          <button onClick={handleSendCode} className="text-sm text-brand-primary hover:underline font-body">
            Resend code
          </button>
        </>
      )}
    </div>
  )
}
```

Profile step, skills step, and activation step follow Women's Portal patterns but adapted for student context (no privacy banner needed to same degree, but keep privacy-respectful language).

Same pattern as Women's Portal auth store but with StudentProfile type:
```typescript
import { create } from 'zustand'
import { persist, createJSONStorage } from 'zustand/middleware'
import type { StudentProfile } from '@glimmora/types'

interface AuthState {
  user: StudentProfile | null
  isAuthenticated: boolean
  onboardingStep: number
  onboardingCompleted: boolean
  setUser: (user: StudentProfile) => void
  setOnboardingStep: (step: number) => void
  completeOnboarding: () => void
  logout: () => void
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set) => ({
      user: null, isAuthenticated: false, onboardingStep: 0, onboardingCompleted: false,
      setUser: (user) => set({ user, isAuthenticated: true }),
      setOnboardingStep: (step) => set({ onboardingStep: step }),
      completeOnboarding: () => set({ onboardingCompleted: true, onboardingStep: 4 }),
      logout: () => set({ user: null, isAuthenticated: false, onboardingStep: 0, onboardingCompleted: false }),
    }),
    { name: 'glimmora-university-auth', storage: createJSONStorage(() => localStorage) }
  )
)
```

```typescript
async function handleExportPDF(credential: PoDLCredential) {
  const { pdf } = await import('@react-pdf/renderer')
  const { PoDLPDFDocument } = await import('./podl-pdf-document')
  const blob = await pdf(<PoDLPDFDocument credential={credential} />).toBlob()
  const url = URL.createObjectURL(blob)
  const a = document.createElement('a')
  a.href = url
  a.download = `podl-${credential.id}.pdf`
  a.click()
  URL.revokeObjectURL(url)
}
```

CRITICAL: Both imports are dynamic (`await import(...)`) to avoid SSR issues.
